import pandas as pd
from scipy import stats


# Filter May & June 
may_data = df[df['Date'].between('2023-05-01', '2023-05-31')]  
june_data = df[df['Date'].between('2023-06-01', '2023-06-30')]



# State null and alternate hypotheses
null_hypothesis = 'The mean temperature in May equals the mean temperature in June'
alt_hypothesis = 'The mean temperature in June is greater than the mean temperature in May'  

# Get February and March temps
may_temps = may_data['Temperature'] 
june_temps = june_data['Temperature']

# Perform independent 2-sample t-test 
t_statistic, p_value = stats.ttest_ind(may_temps, june_temps, equal_var=False)  

# Set significance level 
significance = 0.05

# Interpret p-value
if p_value < significance:
    print('Reject null hypothesis - temperatures means differ significantly')
    print('Accept alternate hypothesis that June temperature is higher')
else:
    print('Fail to reject null hypothesis - no evidence of significant difference')

from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error


# Filter May data
may_data = df[df['Date'].between('2023-05-01', '2023-05-31')]  


# Define X and y variables
X = may_data[['Humidity (percentage)','Rain Trace']]
y = may_data[['Temperature']]  

# Train-test split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=1)

# Train model
model = LinearRegression()
model.fit(X_train, y_train)  

# Evaluate on test data
predictions = model.predict(X_test)
mse = mean_squared_error(y_test, predictions)
print('Test MSE:', mse)



# Get June data for predictions

# Compare actual and predicted
